/* 차트 전역 적용 */
Highcharts.setOptions({
    lang: {
        thousandsSep: ',' // 1000 단위로 , 추가. 선언 후에 개별적으로 format변경 필수
    },
    chart: {
        style: {
            fontFamily: 'pretendard' // 차트 전체 font family 적용
        }
    }
});

Highcharts.chart('차트id', {
    chart: {
        type: '차트타입', // 차트 type지정. bar(y축 기준 bar 차트), column(x축 기준 bar 차트), pie, line, solidgauge, area 등
        height: '400', // 차트 높이 지정
        backgroundColor: 'transparent', // 차트 background color (기본값 #fff),
        marginTop: 50 // 차트 상,하,좌,우 여백 지정. marginTop, marginBottom, marginLeft, marginRight 따로 지정
    },
    navigation: false, // 차트 메뉴 버튼 유/무 (필수)
    credits: {
        enabled: false // 하단 하이차트 주소 숨김 (필수)
    },
    title: { // 대제목, 없을 경우 -> title: undefined or null
        text: 'title'
    },
    subtitle: { // 소제목, 없을 경우 -> subtitle: undefined or null
        text: 'subtitle'
    },
    /* xAxis, yAxis는 pie, solidgauge등 특수한 모양의 차트 제외 적용 */
    xAxis: {
        categories: [
            'x축 카테고리 이름1',
            'x축 카테고리 이름2'
        ],
        crosshair: true, // 가리킨 지점에 background color 넣기. line chart에 적용시 가리킨 데이터에 세로로 선이 그려짐 (기본값 false)
        labels: { // x축 label 설정. yAxis 에도 동일한 옵션 사용
            format: '{value} 월', // label format 설정. value에 1000 단위로 , 추가할 시 {value:,.0f}로 쓰기
            style: { // label 스타일 설정
                color: '#000',
                fontSize: '14px',
                fontWeight: '500',
                transform: 'translate(0, 0)', // x축 이름이 길어 rotate될때 강제로 rotate 막기
            },
        },
        lineColor: '#d6d6d6', // x축 가장 하단 line color 지정
        tickLength: 7, // x축 tick height값
        tickWidth: 2 // x축 tick width값
    },
    yAxis: {
        min: 0, // y축 최소값 (기본값 undefined)
        max: 0, // y축 최대값 (기본값 undefined), max는 보통 잘 적용하지 않음
        gridLineWidth: 0, // y축 그리드 선(가로선) 굵기 (기본값 1)
        title: {
            text: 'title' // y축 제목, 없을 경우 -> title: undefined or null
        },
        gridLineColor: '#d6d6d6', // y축 그리드 color 지정
        tickInterval: 20, // y축 값 간격
        crosshair: true, // 가리킨 지점에 따라가는 선 그리기. (기본값 false)
        opposite: true, // y축을 2개 사용하는 경우, 해당 축을 반대쪽에 표시할지 여부 (기본값 false)
        /* stackLabels는 stack chart일때만 해당 */
        stackLabels: { // stack 차트 상단에 데이터의 합 표출
            enabled: true, // 기본값 false
            style: {
                fontWeight: 'bold',
                color: '#fff',
                textOutline: 'none'
            }
        }
    },
    legend: {
        enabled: false, // legend 없애기. (기본값 true)
        layout: 'vertical', // legend 아이템 정렬방식. horizontal(기본값), vertical
        align: 'left', // legend x축 기준 기본위치. left, center(기본값), right
        verticalAlign: 'top', // legend y축 기준 기본위치. top, middle, bottom(기본값)
        x: 90, // x축 기준 위치
        y: 50, // y축 기준 위치
        backgroundColor: '#fff', // legend 배경색
        borderColor: '#CCC', // legend border색
        borderWidth: 1, // legend border 굵기
        itemStyle: { // legend 아이템 스타일
            fontWeight: '500',
            fontSize: '14px',
            color: '#222429'
        },
        itemDistance: 10, // legend 아이템 사이 간격 
        itemMarginTop: 10, // legend가 vertical일때 아이템 사이 간격
        itemHoverStyle: { // legend 아이템 hover 스타일
            color: '#222429'
        },
        symbolWidth: 8, // legend symbol 가로사이즈
        symbolHeight: 8,  // legend symbol 세로사이즈
        symbolRadius: 0, // legend symbol 둥근정도
        floating: true // 공간을 차지하는 여부, 기본값 false (차지함)
    },
    tooltip: {
        shared: true, // series 각 툴팁을 한 개의 툴팁으로
        useHTML: true, // html 사용 여부
        headerFormat: '<div><b style="color: #000;">{point.key}</b></div>',
        pointFormat: '<b>{point.y}원</b>', // 툴팁 format 지정
        style: { // tooltip 스타일 지정
            color: '#058DC7',
            fontWeight: 'bold'
        }
    },
    plotOptions: {
        // column, bar 적용 옵션 동일
        series: {
            stacking: true, // 기본값 undefined. stack 차트를 만들때만 true 설정. 혹은 단일 series를 사용하는데 dataLabel을 상하 center 정렬 할때 사용. ! column chart 인데 series가 여러개일 경우 사용불가 !
            pointPadding: 0.2, // 막대 사이 padding (기본값 0.1 / ex 0 = 막대 사이 빈 공간 x, 숫자가 커질수록 막대 얇아짐)
            pointWidth: 50, // 막대 width 지정
            borderWidth: 0, // 막대 border 값
            opacity: 0.7, // 차트영역 opacity 값
            borderColor: 'black', // 막대 border 색상 (적용시 borderWidth가 0보다 커야함)
            // border-radius 주려면 highcharts-border-radius.js 필수 (borderRadiusTopLeft / TopRight / BottomLeft / BottomRight)
            // 각 막대별로 다르게 주려면 series에 추가
            borderRadiusTopLeft: 5,
            borderRadiusTopRight: 5,
            dataLabels: {
                enabled: true, // 각 data에 label 추가, 기본값 false
                format: '{y:,.0f}개', // data label format 설정 
                y: 0, // data label 위치 설정 
            },
            label: {
                enabled: false, // 차트영역에 나오는 series name 숨기기, 기본값 true
            },
        }
    },
    colors: ['#058DC7', '#50B432', '#ED561B', '#DDDF00'], // series의 컬러값 지정방법 1번. series 순서에 맞게 컬러값 넣기
    series: [{
        name: '데이터1 이름',
        data: [49, 71], // 특수형 제외 데이타의 개수 = x축 카데고리 개수 같아야함.
        color: '#058DC7', // series의 컬러값 지정방법 2번. 각 series 안에 컬러값 직접 지정
        yAxis: 1, // 다중 y축 사용 시 해당 series의 기준 y축 지정 (기본값 0 (왼쪽))
        /* marker, dashstyle은 line chart일 경우에만 해당 */
        marker: {
            enabled: false, // 포인트 마커 활성화 여부 (기본값 undefined)
            symbol: 'circle', // 심볼지정. 'circle', 'square','diamond', 'triangle', 'triangle-down'
            fillColor: '#FFFFFF', // 심볼 색상
            lineWidth: 3, // 심볼 border width
            lineColor: '#e34d27', // 심볼 border color
            radius: 4 // 심볼 사이즈
        },
        dashStyle: 'shortdot' // line chart일때 선 스타일
    }, {
        name: '데이터2 이름',
        data: [83, 78]
    }, {
        name: '데이터3 이름',
        data: [48, 38]
    }, {
        name: '데이터4 이름',
        data: [42, 33]
    }],
    responsive: { // 차트 반응형 설정
        rules: [{
            condition: {
                maxWidth: 500 // 차트 크기가 500px 이하일때 적용
            },
            chartOptions: {
                // xAxis, yAxis, legend 등 옵션들 수정
            }
        }]
    }
});

var gaugeOptions = {
    chart: {
        type: 'solidgauge',
    },
    title: { // 없을 경우 -> title: undefined or null
        text: 'title',
        y: 50
    },
    pane: {
        startAngle: -90, // 시작 각도 위치 (기본값 0)
        endAngle: 90, // 끝 각도 위치 (기본값 0)
        background: {
            backgroundColor: '#eee',
            innerRadius: '60%', // 원 내부 반경 (가운데 구멍 크기) (기본값 0)
            outerRadius: '100%', // 원 외부 반경 (기본값 105%)
            shape: 'arc'
        }
    },
    exporting: {
        enabled: false // 내보내기 활성화 여부 (기본값 true) (navigation: false와 비슷)
    },
    credits: {
        enabled: false
    },
    tooltip: {
        enabled: false // 툴팁 활성화 여부 (기본값 true)
    },
    yAxis: {
        stops: [ // 게이지 값 별 색상(그라디언트 형식) 정의 (solidgauge만 해당)
            [0.1, '#e14159'],
            [0.3, '#f99d42'],
            [0.5, '#facf4c'],
            [0.7, '#83ab47'],
            [0.9, '#5587a2']
        ],
        lineWidth: 0, // 선 너비 (기본값 1)
        tickWidth: 0, // 눈금 표시 픽셀 너비 (기본값 undefined)
        tickAmount: 2, // 눈금 개수 (기본값 undefined)
        minorTickInterval: undefined, // 눈금 사이 보조 눈금 간격 (기본값 undefined)
        labels: {
            y: 16
        }
    },
    plotOptions: {
        solidgauge: {
            dataLabels: {
                y: -40,
                borderWidth: 0, // 기본값 1
                useHTML: true // html 사용 여부
            }
        }
    }
};

var solidgauge = Highcharts.chart('solidgaugeChart', Highcharts.merge(gaugeOptions, {

    },
    series: [{
        name: 'match',
        data: [50],
        dataLabels: {
            format: '<span style="font-size: 28px;">{y}</span>'
        }
    }]
}));
