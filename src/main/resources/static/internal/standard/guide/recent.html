// tab
$(document).ready(function () {
    if ($('.js_tmenu').length || $('.js_tcont').length) {
        $('.js_tcont').stop().hide();
        $('.js_tcont.on').stop().show();
        $('.js_tmenu li a').click(function () {
            let tabId = $(this).parent().attr('id');
            let selTabId = $('.js_tmenu li[id="' + tabId + '"], .js_tcont[data-tab="' + tabId + '"]');
            $(this).closest('.tab').find('.js_tmenu li, .js_tcont').not('.js_tmenu li.on').removeClass('on');
            selTabId.addClass('on').fadeIn();
            selTabId.siblings('.js_tcont').stop().hide();
            selTabId.siblings().removeClass('on');
            return false;
        });
    }
});

$(document).ready(function () {
    if ($('.js_tmenu').length || $('.js_tcont').length) {
        $('.js_tcont').stop().hide();
        $('.js_tcont.on').stop().show();
        $('.js_tmenu li a').click(function () {
            let tabId = $(this).parent().attr('id');
            let selTabId = $('.js_tmenu li[id="' + tabId + '"], .js_tcont[data-tab="' + tabId + '"]');
            $(this).closest('.tab').find('.js_tmenu li, .js_tcont').not('.js_tmenu li.on').removeClass('on');
            selTabId.addClass('on').fadeIn();
            selTabId.siblings('.js_tcont').stop().hide();
            selTabId.siblings().removeClass('on');
            return false;
        });
    }
});

// recent slider
$(document).ready(function () {
	function slider() {
		var swiperArr = [];
		var slideInx = [];
		var slideNum = [];
		$(".recent_slider").each(function (index) {
			var $this = $(this);
			$this.addClass("slider-" + index);
			swiperArr.push(undefined);
			slideInx.push(0);
			slideNum.push($this.find('.swiper-slide').length);
		});
		function sliderAct() {
			$(".recent_slider").each(function (index) {
				if (swiperArr[index] != undefined) {
					swiperArr[index].destroy();
					swiperArr[index] = undefined;
				}
				swiperArr[index] = new Swiper('.slider-' + index + ' .swiper-container', {
					slidesPerView: 'auto',
					loop: true,
					speed: 500,
					observer: true,
					observeParents: true,
					navigation: {
						nextEl: $('.slider-' + index).find('.btn_recent_next'),
						prevEl: $('.slider-' + index).find('.btn_recent_prev')
					},
					on: {
						activeIndexChange: function () {
							if ($('.slider-' + index).parent('.tab_cont').css('display') != 'none') {
								slideInx[index] = this.realIndex;
							}
						}
					},
				});
				if (swiperArr[index] == undefined) {
					swiperArr[index] = swiper;
				}
			});
		}
		sliderAct();
	}
	slider();
});
