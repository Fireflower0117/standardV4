레이어 팝업

1. 닫기 버튼 or 배경을 클릭하면 팝업이 닫힌다.
2. 팝업이 여러 개 생성될 경우 view_show와 display_view의 숫자가 동일해야 한다.
3. popup_bg는 한 번만 넣어 처리한다.
4. 팝업 넓이 조정: js_popup 옆 클래스 w(숫자)px, w(숫자)p 적용
ex) ... js_popup w500px">
5. 팝업의 컨텐츠 영역에는 max-height값을 지정해 일정 높이 이상일때 스크롤이 생기도록 한다. 이때, simplebar.js 플러그인이 자동 적용 되도록 한다. (기본 스크롤 스타일 적용시, 플러그인 적용 안해도 ok)

/** popup **/
// popup - layer
function view_show(popName) {
    let left = ($(window).width() - $("#display_view" + popName).width()) / 2;
    let top = ($(window).height() - $("#display_view" + popName).height()) / 2;
    $("#js_popup_bg").show();
    $("#display_view" + popName).addClass("on").css({ 'visibility': 'visible', 'position': 'fixed', 'left': left, 'top': top, 'z-index': 5500 });
    return false;
}
function view_hide(popName) {
    $("#display_view" + popName).removeClass("on").css('visibility', 'hidden');
    $("#js_popup_bg").hide();
    return false;
}
$(function () {
    $('#js_popup_bg').click(function () {
        $('.js_popup').removeClass("on").css('visibility', 'hidden');
        $(this).hide();
    });
});
/** // popup **/
