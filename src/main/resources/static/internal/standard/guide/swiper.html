swiper01

타이틀
Slide 1
Slide 2
Slide 3

코드

  • html
  • css
<div class="demoSwiper01">
    <div class="tit">
        <strong>타이틀</strong>
        <div class="navi">
            <button class="btn ic"><i class="xi-angle-left"></i></button>
            <button class="btn ic"><i class="xi-pause"></i></button>
            <button class="btn ic"><i class="xi-angle-right"></i></button>
        </div>
    </div>
    <div class="swiper-wrapper">
        <div class="swiper-slide">Slide 1</div>
        <div class="swiper-slide">Slide 2</div>
        <div class="swiper-slide">Slide 3</div>
    </div>
</div>
<script>
    var demoSwiper01 = new Swiper(".demoSwiper01", {
        speed: 1000, // 슬라이드 전환 속도 (기본 300(ms))
        loop: true,
        autoplay: {
            delay: 2000, // 슬라이드 지연 속도 (기본 3000(ms))
            disableOnInteraction: false // 직접 스와이프 후 자동재생 비활성화 여부 (기본 true)
        },
        navigation: {
            prevEl: ".xi-angle-left",
            nextEl: ".xi-angle-right",
        },
    });
    $('.demoSwiper01 .navi .xi-pause').click(function () {
        if($(this).hasClass('xi-play')) {
            demoSwiper01.autoplay.start();
            $(this).removeClass('xi-play').addClass('xi-pause');
        } else {
            demoSwiper01.autoplay.stop();
            $(this).removeClass('xi-pause').addClass('xi-play');
        }
    });
</script>

        <div class="swiper-slide"><div>Slide 2</div></div>
        <div class="swiper-slide"><div>Slide 3</div></div>
        <div class="swiper-slide"><div>Slide 4</div></div>
        <div class="swiper-slide"><div>Slide 5</div></div>
        <div class="swiper-slide"><div>Slide 6</div></div>
    </div>
</div>
<script>
    var demoSwiper04 = new Swiper(".demoSwiper04", {
        slidesPerView: 5, // 한 번에 보이는 슬라이드 개수 (기본 1)
        spaceBetween: 30, // 슬라이드 사이 거리 (기본 0(px))
        speed: 1000, // 슬라이드 전환 속도 (기본 300(ms))
        centeredSlides: true, // 활성 슬라이드 중앙 배치 여부 (기본 false)
        effect: 'coverflow', // 커버플로우 효과
        coverflowEffect: {
            rotate: 0, // 슬라이드 회전 각도 (기본 50)
            slideShadows: false, // 슬라이드 그림자 활성화 (기본 true)
        },
        loop: true,
        autoplay: {
            delay: 2000, // 슬라이드 지연 속도 (기본 3000(ms))
            disableOnInteraction: false // 직접 스와이프 후 자동재생 비활성화 여부 (기본 true)
        },
    });
</script>
