basic.zip 기본요소 종합 파일


팝업

  • 레이어 팝업
  • 메인 팝업

Default

1. mobile_basic.css를 반드시 먼저 import 해야함. (mobile.zip 구성요소 모두 필수)
2. mobile_basic.css import 이후 css 개별 추가, 혹은 종합 스타일이 들어있는 mobile.css를 import 해서 사용 (mobile.css import 시 mobile_basic.css와 css 개별 추가 필요X)

레이어 팝업

1. 닫기 버튼 or 배경을 클릭하면 팝업이 닫힌다.
2. 팝업이 여러 개 생성될 경우 view_show와 display_view의 숫자가 동일해야 한다.
3. popup_bg는 한 번만 넣어 처리한다.
4. 팝업 넓이 : 95%

/** popup **/
// popup - layer
function view_show(popName) {
    let top = ($(window).height() - $("#display_view" + popName).height()) / 2;
    $("#js_popup_bg").show();
    $("#display_view" + popName).addClass("on").css({ 'visibility': 'visible', 'position': 'fixed', 'top': top, 'z-index': 5500 });
    return false;
}
function view_hide(popName) {
    $("#display_view" + popName).removeClass("on").css('visibility', 'hidden');
    $("#js_popup_bg").hide();
    return false;
}
$(function () {
    $('#js_popup_bg').click(function () {
        $('.js_popup').removeClass("on").css('visibility', 'hidden');
        $(this).hide();
    });
});
/** // popup **/
