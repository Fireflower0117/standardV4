* 방향키로 스크롤 조작 가능, shift키로 가로 스크롤 동작 불가
- data-simplebar 속성을 사용하면 따로 js를 추가하지 않아도 스크롤이 적용된다.
- 가상의 dom을 만들어 구조가 바뀌는 플러그인 이므로, ul, table 등과 같이 태그 바로 하위에 div가 오면 안 되는 태그들은 div로 감싸서 사용한다.
ex) <div data-simplebar style="max-height: 150px;"><ul></ul></div>
- 기본 속성이 auto hide(스크롤 영역에 마우스가 없을때 스크롤바가 투명해짐) 이므로, 스크롤이 항상 보이게 하려면 data-simplebar-auto-hide="false" 속성을 추가한다.

document.addEventListener('DOMContentLoaded', function () {
    var scrollwrappers = document.querySelectorAll('.scrwrap');
    scrollwrappers.forEach(function (wrapper) {
        var scrollUpButton = wrapper.querySelector('.scroll_up_button');
        var scrollDownButton = wrapper.querySelector('.scroll_down_button');
        var scrollInterval;
        if (scrollUpButton && scrollDownButton) {
            scrollUpButton.addEventListener('mousedown', function () {
                scrollInterval = setInterval(function () {
                    wrapper.querySelector('.simplebar-content-wrapper').scrollTop -= 5; // 스크롤을 위로 이동시키는 값 조절
                }, 10); // 스크롤 속도 조절
            });
            scrollDownButton.addEventListener('mousedown', function () {
                scrollInterval = setInterval(function () {
                    wrapper.querySelector('.simplebar-content-wrapper').scrollTop += 5; // 스크롤을 아래로 이동시키는 값 조절
                }, 10); // 스크롤 속도 조절
            });
            // 마우스 버튼을 놓으면 스크롤 중지
            document.addEventListener('mouseup', function () {
                clearInterval(scrollInterval);
            });
        }
    });
});
